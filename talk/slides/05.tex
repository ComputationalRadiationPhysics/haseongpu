\begin{frame}{Monte Carlo integration is a scalable simulation method}
  \begin{columns}[T]

    \begin{column}{.5\textwidth}
      \textbf{graphic:integration transformed into MC}
    \end{column}

    \begin{column}{.5\textwidth}
      \begin{itemize}
        \item Monte Carlo integration can be transformed into a sum
        \myuncover{2}{3}{
          \begin{equation}
            \int_{N}{f(x) dx} \Rightarrow \sum^{N}_{i=1}{f(i)}
          \end{equation}
        }
        \myuncover{3}{3}{
        \item All elements of the sum are mutually independent, which is great
          for parallelization
        }
      \end{itemize}
    \end{column}

  \end{columns}
\end{frame}
